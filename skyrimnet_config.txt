# CFG_SCALE: How Closely to Follow the Recipe
# Default: A great starting point.
# Voice sounds strained or weird? Lower this value. It tells the model to be more relaxed 
# and improvisational, great for expressive prompts.
# Need maximum clarity and adherence to the text? Raise it slightly to keep the model on a tighter leash.
cfg_scale = default

# INFERENCE_TIMESTEPS: Simmering Time (Quality vs. Speed)
# Need a quick snack? Use a lower number. Perfect for fast drafts and experiments.
# Cooking a gourmet meal? Use a higher number. This lets the model "simmer" longer, 
# refining the audio for superior detail and naturalness.
# IMPORTANT: This uses the "confidence" slider in SkyrimNet UI when set to "api"
inference_timesteps = default
